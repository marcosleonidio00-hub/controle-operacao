  <script>
    const REGISTRO_FIELDS = {};

    function mostrarFeedback(msg, tipo) {
      const f = document.getElementById('feedback');
      f.textContent = msg;
      f.className = tipo;
      f.style.display = 'block';
      setTimeout(()=> f.style.display='none', 5000);
    }

    function showLoading(show=true) {
      document.getElementById('loadingOverlay').style.display = show?'flex':'none';
    }

    function limparFormulario() {
      document.getElementById('numPedidoBusca').value='';
      document.getElementById('dadosPedido').style.display='none';
      document.getElementById('btnRegistrar').disabled=true;
      document.getElementById('regPax').value='';
      document.getElementById('regMotivo').value='';
      for(const k in REGISTRO_FIELDS) delete REGISTRO_FIELDS[k];
    }

    function buscarDados() {
      const numPedido = document.getElementById('numPedidoBusca').value.trim();
      if(!numPedido){ mostrarFeedback('Digite o nº do pedido', 'error'); return; }
      showLoading(true);
      google.script.run
        .withSuccessHandler(dados => { showLoading(false); exibirDadosParaRegistro(dados); })
        .withFailureHandler(err => { showLoading(false); mostrarFeedback('Erro: '+err, 'error'); })
        .buscarPedido(numPedido);
    }

    function exibirDadosParaRegistro(dados){
      if(!dados){ mostrarFeedback('Pedido não encontrado', 'error'); return; }
      REGISTRO_FIELDS.numPedido=dados.numPedido;
      REGISTRO_FIELDS.fornecedor=dados.fornecedor;
      REGISTRO_FIELDS.atracao=dados.atracao;
      REGISTRO_FIELDS.booking=dados.booking;
      REGISTRO_FIELDS.valor=dados.valor;
      REGISTRO_FIELDS.criadoPor=dados.criadoPor;

      document.getElementById('regNumPedido').value=dados.numPedido||'';
      document.getElementById('regFornecedor').value=dados.fornecedor||'';
      document.getElementById('regAtracao').value=dados.atracao||'';
      document.getElementById('regBooking').value=dados.booking||'';
      document.getElementById('regValor').value=dados.valor||'';
      document.getElementById('regCriadoPor').value=dados.criadoPor||'';

      document.getElementById('dadosPedido').style.display='block';
      document.getElementById('btnRegistrar').disabled=false;
      mostrarFeedback('Pedido encontrado. Preencha Pax e Motivo.', 'success');
    }

    function registrar() {
      const pax = document.getElementById('regPax').value.trim();
      const motivo = document.getElementById('regMotivo').value.trim();
      if(!pax||!motivo){ mostrarFeedback('Preencha Pax e Motivo', 'error'); return; }
      const formData = {...REGISTRO_FIELDS, pax, motivo};
      showLoading(true);
      google.script.run
        .withSuccessHandler(res=>{ showLoading(false); mostrarFeedback(res.mensagem,res.sucesso?'success':'error'); if(res.sucesso){ limparFormulario(); exibirAjustes(); } })
        .withFailureHandler(err=>{ showLoading(false); mostrarFeedback('Erro: '+err,'error'); })
        .registrarCancelamento(formData);
    }

    function abrirPainelAlteracao(){
      const panel = document.getElementById('alteracao-panel');
      panel.style.display = panel.style.display==='block'?'none':'block';
      if(panel.style.display==='block') exibirAjustes();
    }

    function exibirAjustes(mostrarTodos=false){
      const filtro = mostrarTodos?null:document.getElementById('numPedidoAlteracao').value.trim();
      const lista = document.getElementById('lista-ajustes');
      lista.innerHTML='<p>Carregando ajustes...</p>';
      google.script.run
        .withSuccessHandler(buildAlteracaoPanel)
        .withFailureHandler(err=>lista.innerHTML='<p class="error">Erro: '+err+'</p>')
        .buscarAjustes(filtro);
    }

    function buildAlteracaoPanel(ajustes){
      const lista = document.getElementById('lista-ajustes');
      if(!ajustes||ajustes.length===0){ lista.innerHTML='<p>Nenhum ajuste encontrado.</p>'; return; }
      let html=`<h4>Total: ${ajustes.length}</h4>`;
      ajustes.forEach(a=>{
        html+=`
        <div class="alteracao-item" data-row-index="${a.rowIndex}">
          <p style="font-weight:bold; color:#a80000;">Pedido: ${a.N_PEDIDO||''} | Status: ${a.STATUS||''} (Linha ${a.rowIndex})</p>
          <label>Status:</label>
          <select id="status-${a.rowIndex}">
            <option value="PENDENTE ALTERACAO" ${a.STATUS==='PENDENTE ALTERACAO'?'selected':''}>PENDENTE ALTERACAO</option>
            <option value="RESOLVIDO" ${a.STATUS==='RESOLVIDO'?'selected':''}>RESOLVIDO</option>
          </select>
          <label>Pax:</label>
          <input type="text" id="pax-${a.rowIndex}" value="${a.PAX||''}">
          <label>Motivo:</label>
          <textarea id="motivo-${a.rowIndex}" rows="1">${a.MOTIVO||''}</textarea>
          <button class="btn-registrar" onclick="salvarAlteracao(${a.rowIndex})">Salvar Alteração</button>
          <span id="update-feedback-${a.rowIndex}"></span>
        </div>`;
      });
      lista.innerHTML=html;
    }

    function salvarAlteracao(rowIndex){
      const updateData = {
        rowIndex,
        Status: document.getElementById(`status-${rowIndex}`).value,
        Pax: document.getElementById(`pax-${rowIndex}`).value,
        Motivo: document.getElementById(`motivo-${rowIndex}`).value
      };
      const feedback = document.getElementById(`update-feedback-${rowIndex}`);
      feedback.innerHTML='Salvando...';
      google.script.run
        .withSuccessHandler(res=>{ exibirAjustes(true); mostrarFeedback(res.mensagem,res.sucesso?'success':'error'); })
        .withFailureHandler(err=>feedback.innerHTML='<span class="error">Erro: '+err+'</span>')
        .atualizarAjuste(updateData);
    }

  </script>
